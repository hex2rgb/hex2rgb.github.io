Numpy取值

2020-09-13 · 69 个字 · 1 分钟 读完
# coding=utf-8
import numpy as np
# ./np.csv
# 1,2,3,4
# 5,6,7,8
# 9,8,7,6
# 5,4,3,7
# 2,6,8,9
# 12,61,18,19
t1 = np.loadtxt("./np.csv", delimiter=",", dtype="int")
print(t1)
# 取行
print(t1[2])
# 取多行
print(t1[2:])
# 取不连续多行
print(t1[[2, 3, 5]])

# 取列
print(t1[:, 0])
# 取连续多列
print(t1[:, 1:])
# 取不连续多列
print(t1[:, [0, 2, 3]])

# 取第3行，第2列
a = t1[3, 2]
print(a)
print(type(a))

# 取第1-3行，第1-3列
print(t1[0:3, 0:3])

# 取不相邻的点
print(t1[[0, 1], [2, 1]])
